# UTILITIES-X

Plugin con opciones básicas para nuestro servidor de PocketMine.
Como por ejemplo un /fly, /gm, /hub entre más!

# Commands

    fly:
      permission: "fly.cmd"
      use: "/fly on off"
    ping:
      permission: "ping.cmd"
      use: "/ping"

    gm:
      permission: "gm.cmd"
      use: "/gm 0 1 3"

    hub:
      permission: "hub.cmd"
      use: "/hub"

    utils:
      permission: "utils.cmd"
      use: "/showutils"

    sethub:
      permission: "set.hub"
      use: "/sethub set <map>"
